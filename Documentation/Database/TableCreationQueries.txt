Below are the queries that are used to create all of the tables

create table customers(custID int auto_increment, fname varchar(50), lname varchar(50), username varchar(50), password varchar(50), email varchar(50), primary key(custID));
create table admins(adminid int auto_increment, fname varchar(50), lname varchar(50), username varchar(50), password varchar(50), primary key(adminid));
create table orders(orderid int auto_increment, custID int, isCustomOrder varchar(2), amount double, date date, transID int, status varchar(50), primary key(orderid));
create table transactions(transID int auto_increment, date date, balance double, primary key(transid));
create table custom_order_details(custorderdetailid int auto_increment, itemid int, orderDesription varchar(50), orderid int, primary key(custorderdetailid));
create table non_custom_order_details(orderdetailid int auto_increment, itemid int, orderDesription varchar(50), orderid int, primary key(orderdetailid));
create table items(itemid int auto_increment, itemprice double, itemname varchar(255), itemLength int, itemWidth int, primary key(itemid));
create table raw_materials(rawmaterialid int auto_increment, itemID int, materialName varchar(255), amountInStock int, inStock varchar(2), primary key(rawmaterialid));
